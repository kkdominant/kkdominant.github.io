不定积分

计算下面不定积分

$$
\begin{align*}
& \left( 1 \right) \int{\sqrt{\frac{a-bx}{a+bx}}dx} \
& \left( 2 \right) \int{\frac{x+\sin x}{1+\cos x}dx}\
&\left( 3 \right) \int{\frac{\cos x+x\sin x}{\left( x+\cos x \right) ^2}dx}\

&\left( 4 \right)\ \int{\frac{x^2}{\left( x\cos x-\sin x \right) ^2}dx}\
& \left( 5 \right) \int{\frac{x+\sin x\cos x}{\left( \cos x-x\sin x \right) ^2}dx}\
& \left( 6 \right) \int{\frac{x^2}{\left( x\cos x-\sin x \right) \left( x\sin x+\cos x \right)}}dx\
&\left( 7 \right)\int{\left( 1+x-\frac{1}{x} \right) e^{x+\frac{1}{x}}dx}

\end{align*}
$$

参考解答：

$$
\begin{aligned}
&\ \left( 1 \right) :\int{\sqrt{\frac{a-bx}{a+bx}}dx} &
\
&\ =:\int{\frac{a-bx}{\sqrt{a^2-b^2x^2}}dx}&
\
&\ \xrightarrow{x=\frac{a}{b}\sin t}:\int{\frac{a-\mathrm{a}\sin t}{\mathrm{a}\cos t}.\frac{a}{b}\cos t}dt&
\
&\ =\frac{a}{b}\left( t+\cos t \right) ,,+C,, \text{由}x=\frac{a}{b}\sin t\text{得：}t=\mathrm{arc}\sin \frac{b}{a}x\text{，}\cos t=\frac{\sqrt{a^2-b^2x^2}}{a} &
\
&\ =\frac{a}{b}\mathrm{arc}\sin \frac{b}{a}x+\frac{\sqrt{a^2-b^2x^2}}{b}+C &
\
&\ \text{故}\int{\sqrt{\frac{3-2x}{3+2x}}dx=\frac{3}{2}\mathrm{arc}\sin \frac{2}{3}x+\frac{\sqrt{9-4x^2}}{2}+C}&
\
&\ \left( 2 \right) \int{\frac{x+\sin x}{1+\cos x}dx}&
\
&\ =\int{\frac{x}{1+\cos x}dx+\int{\frac{\sin x}{1+\cos x}dx}}&
\
&\ =\int{\frac{x}{2}\sec ^2\frac{x}{2}dx-\ln \left| 1+\cos x \right|}&
\
&\ =x\tan \frac{x}{2}-\ln \left| 1+\cos x \right|+2\ln \left| \cos \frac{x}{2} \right|+C&
\
&\
\left( 3 \right)

\int{\frac{\cos x+x\sin x}{\left( x+\cos x \right) ^2}dx}&
\
&\ =\int{\frac{\cos x+x}{\left( x+\cos x \right) ^2}dx}+\int{\frac{x\left( \sin x-x \right)}{\left( x+\cos x \right) ^2}dx}
\
&\ =\int{\frac{1}{x+\cos x}dx}-\int{\frac{x}{\left( x+\cos x \right) ^2}d\left( \cos x+x \right)}
\
&\ =\int{\frac{1}{x+\cos x}dx}+\int{xd\frac{1}{\left( \cos x+x \right)}}
\
&\ =\int{\frac{1}{x+\cos x}dx}+\frac{x}{x+\cos x}-\int{\frac{1}{x+\cos x}dx}
\
&\ =\frac{x}{x+\cos x}+C
\

&\ \left( 4 \right) \because d\frac{1}{x\cos x-\sin x}=\frac{-x\sin x}{\left( x\cos x-\sin x \right) ^2}dx &
\
&\ \int{\frac{x^2}{\left( x\cos x-\sin x \right) ^2}dx}
\
&\ =-\int{\frac{x}{\sin x}\frac{x\sin x}{\left( x\cos x-\sin x \right) ^2}dx}
\
&\ =\int{\frac{x}{\sin x}d\frac{1}{x\cos x-\sin x}}
\
&\ =\frac{x}{x\sin x\cos x-\sin ^2x}-\int{\frac{1}{x\cos x-\sin x}d\frac{x}{\sin x}}
\
&\ =\frac{x}{x\sin x\cos x-\sin ^2x}-\int{\frac{1}{x\cos x-\sin x}.\frac{\sin x-x\cos x}{\sin ^2x}dx}
\
&\ =\frac{x}{x\sin x\cos x-\sin ^2x}-\cot x+C
\
&\ \left( 5 \right) \int{\frac{x+\sin x\cos x}{\left( \cos x-x\sin x \right) ^2}dx}&
\
&\ \xrightarrow{\text{分子分母同除}\cos ^2x}\int{\frac{x\sec ^2x+\tan x}{\left( 1-x\tan x \right) ^2}dx}
\
&\ =\int{\frac{1}{\left( 1-x\tan x \right) ^2}dx\tan x}
\
&\ =\frac{1}{1-x\tan x}+C
\

& \left( 6 \right) \int{\frac{x^2}{\left( x\cos x-\sin x \right) \left( x\sin x+\cos x \right)}}dx&
\
&=\int{\frac{x\sin x}{\left( x\cos x-\sin x \right)}}dx+\int{\frac{x\cos x}{\left( x\sin x+\cos x \right)}}dx&
\
&=\ln \left| \frac{x\sin x+\cos x}{x\cos x-\sin x} \right|+C&
\
&\left( 7 \right) \because de^{x+\frac{1}{x}}=\left( 1-\frac{1}{x^2} \right) e^{x+\frac{1}{x}}dx=\frac{1}{x}\left( x-\frac{1}{x} \right) e^{x+\frac{1}{x}}dx&
\
&\int{\left( 1+x-\frac{1}{x} \right) e^{x+\frac{1}{x}}dx}&
\
&=\int{e^{x+\frac{1}{x}}dx}+\int{x}de^{x+\frac{1}{x}}&
\
&=\int{e^{x+\frac{1}{x}}dx}+xe^{x+\frac{1}{x}}-\int{e^{x+\frac{1}{x}}dx}&
\
&=xe^{x+\frac{1}{x}}+C&
\
\end{aligned}
$$
